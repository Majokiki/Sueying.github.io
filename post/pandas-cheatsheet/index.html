日期：移动，范围，频率，偏移

>>> import pandas as pd
>>> import numpy as np
>>> date = pd.to_datetime('2020-08-01')
>>> date
Timestamp('2020-08-01 00:00:00')
# 移动
>>>date = date - pd.DateOffset(years=1)
>>>date = date - pd.DateOffset(days=7)

# 范围
>>> index = pd.date_range('2020-08-01', '2020-08-30')
>>> index
DatetimeIndex(['2020-08-01', '2020-08-02', '2020-08-03', '2020-08-04',
               '2020-08-05', '2020-08-06', '2020-08-07', '2020-08-08',
               '2020-08-09', '2020-08-10'],
              dtype='datetime64[ns]', freq='D')

>>> pd.date_range(start='2020-08-01',  periods=10)
DatetimeIndex(['2020-08-01', '2020-08-02', '2020-08-03', '2020-08-04',
               '2020-08-05', '2020-08-06', '2020-08-07', '2020-08-08',
               '2020-08-09', '2020-08-10'],
              dtype='datetime64[ns]', freq='D')

# 频率
>>> ts = Series(np.random.randn(4),index=pd.date_range('1/1/2000', periods=4, freq='M'))
>>> ts
2000-01-31    0.556812
2000-02-29    0.004086
2000-03-31   -1.202868
2000-04-30   -0.337943
Freq: M, dtype: float64

# 偏移
>>> ts.shift(2)
2000-01-31         NaN
2000-02-29         NaN
2000-03-31    0.556812
2000-04-30    0.004086
Freq: M, dtype: float64

>>> ts/ts.shift(1)-1 
2000-01-31           NaN
2000-02-29     -0.992662
2000-03-31   -295.398535
2000-04-30     -0.719052
Freq: M, dtype: float64

>>> ts.shift(2, freq='M')
2000-03-31    0.556812
2000-04-30    0.004086
2000-05-31   -1.202868
2000-06-30   -0.337943
Freq: M, dtype: float64
